``` r
gapminder$continent %>%

  levels()
```

    ## [1] "Africa"   "Asia"     "Europe"   "Americas" "Oceania"

``` r
gap2 <- gapminder %>% 
  mutate(continent = fct_infreq(continent))

  levels()
```
